Report call of mutable method by immutable value. For example: 
struct Foo {}
fn (mut f Foo) bar() {}
//   ^^^ f is mutable

fn main() {
    a := Foo{}
//  ^ a is immutable
    a.bar()
//    ^^^^^ call of mutable method by immutable value

    mut b := Foo{}
//      ^ b is mutable
    b.bar()
//    ^^^^^ all ok
}
